tags与categories的3D页面

最后更新: 2025年11月23日 作者: 光田云恒 字数: 476 预计阅读时间: 2min

Tags 页面

<%- partial('partials/header') %>
<div class="wrapper">
  <header class="header">
    <h1 class="header-title center">Tags</h1>
  </header>
  <!-- 3D 标签球 -->
  <div style="text-align:center; margin: 40px 0;">
    <canvas id="tagCanvas" width="450" height="450"></canvas>
    <!-- Tag 列表（供 JS 读取，但隐藏） -->
    <div id="tagList" style="display:none;">
      <% if (site.tags && site.tags.length) { %>
        <% site.tags.sort('name').each(function(tag){ %>
          <a class="tag-item" href="<%- url_for(tag.path || ('/tags/' + tag.name + '/')) %>"><%= tag.name %></a>
        <% }) %>
      <% } %>
    </div>
  </div>
</div>
<script src="<%- url_for('/js/tagcanvas.js') %>"></script>
<script>
function getTagColor() {
  return document.body.getAttribute('data-theme') === 'dark' ? '#eaeaea' : '#333';
}
document.addEventListener('DOMContentLoaded', function () {
  if (typeof TagCanvas === 'undefined') return;
  const items = document.querySelectorAll('#tagList .tag-item');
  if (!items || items.length === 0) return;
  const sphere = new TagCanvas('tagCanvas', {
    radius: 180,
    speed: 0.004,
    font: '18px Arial',
    color: getTagColor()
  });
  const observer = new MutationObserver(function () {
    sphere.color = getTagColor();
  });
  observer.observe(document.body, { attributes: true });
});
</script>
<%- partial('partials/foot') %>
